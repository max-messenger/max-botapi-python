### –ó–∞–ø—É—Å–∫ Polling

–ï—Å–ª–∏ —É –±–æ—Ç–∞ —É—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω—ã –ø–æ–¥–ø–∏—Å–∫–∏ –Ω–∞ Webhook - —Å–æ–±—ã—Ç–∏—è –Ω–µ –±—É–¥—É—Ç –ø—Ä–∏—Ö–æ–¥–∏—Ç—å –ø—Ä–∏ –º–µ—Ç–æ–¥–µ `start_polling`. —á–µ—Ä–µ–∑ `await bot.delete_webhook()` –ø–µ—Ä–µ–¥ `start_polling`.

import logging

from maxapi import Bot, Dispatcher
from maxapi.types import BotStarted, Command, MessageCreated

logging.basicConfig(level=logging.INFO)

bot = Bot('—Ç—É—Ç_–≤–∞—à_—Ç–æ–∫–µ–Ω')
dp = Dispatcher()

# –û—Ç–≤–µ—Ç –±–æ—Ç–∞ –ø—Ä–∏ –Ω–∞–∂–∞—Ç–∏–∏ –Ω–∞ –∫–Ω–æ–ø–∫—É "–ù–∞—á–∞—Ç—å"
@dp.bot_started()
async def bot_started(event: BotStarted):
    await event.bot.send_message(
        chat_id=event.chat_id,
        text='–ü—Ä–∏–≤–µ—Ç! –û—Ç–ø—Ä–∞–≤—å –º–Ω–µ /start'
    )

# –û—Ç–≤–µ—Ç –±–æ—Ç–∞ –Ω–∞ –∫–æ–º–∞–Ω–¥—É /start
@dp.message_created(Command('start'))
async def hello(event: MessageCreated):
    await event.message.answer(f"–ü—Ä–∏–º–µ—Ä —á–∞—Ç-–±–æ—Ç–∞ –¥–ª—è MAX üíô")

    await dp.start_polling(bot)

    asyncio.run(main())
